drop table producer cascade constraints;
create table producer(
	code varchar2(10) primary key,
	pname varchar2(10),
	pprice number
);

drop table consumer cascade constraints;
create table consumer(
	--code varchar2(10) unique references producer(code) on delete set null,
	code varchar2(10) references producer(code) on delete cascade,
	cname varchar2(10),
	cprice number,
	caddr varchar2(10)
);
